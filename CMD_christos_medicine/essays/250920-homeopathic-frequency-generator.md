import numpy as np
import matplotlib.pyplot as plt
from scipy.signal import find_peaks
from ipywidgets import interact, FloatSlider, IntSlider

# Step 1: Simulate substance scan (cymatics dust patterns)
def scan_substance(min_freq=1, max_freq=1e6, num_freq=1000, noise_level=0.5):
    freqs = np.linspace(min_freq, max_freq, num_freq)
    # Simulate pattern order: Coherent peaks at harmonics + noise
    pattern = np.sin(2*np.pi*freqs/100) + 0.5*np.sin(4*np.pi*freqs/100) + noise_level*np.random.randn(num_freq)
    peaks, _ = find_peaks(pattern, height=1)  # Coherent freqs > threshold
    return freqs, pattern, freqs[peaks]

# Step 2: Simulate patient diagnosis (disrupted resonance)
def patient_pattern(coherent_freqs, disruption=0.8):
    # Disrupt: Shift/attenuate freqs
    disrupted = coherent_freqs * (1 - disruption*np.random.rand(len(coherent_freqs)))
    return disrupted

# Step 3: Apply therapy (toroid/AB simulation: Phase shift to 'tip' restoration)
def apply_therapy(disrupted, coherent_freqs, alpha=1.0):
    # Simulate AB stress: Phase shift restores via feedback
    restored = disrupted + alpha * (coherent_freqs - disrupted)
    return restored

# Interactive Viz
def sim_homeopathy(noise_level, disruption, alpha):

    plt.figure(figsize=(12, 6))
    plt.subplot(3,1,1)

    plt.ylabel('Pattern Amplitude')

    plt.stem(disrupted, np.ones_like(disrupted)*0.5, linefmt='r-', markerfmt='ro', label='Disrupted')

    plt.xlabel('Frequency (Hz)')
